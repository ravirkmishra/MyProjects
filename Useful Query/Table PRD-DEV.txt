Step 2:

Then create SP which will insert into the required table.

ALTER PROCEDURE [dbo].[InsertTable]
    @dt MyTableType readonly
AS
BEGIN
	SET IDENTITY_INSERT [dbo].[APROBACIONBackUp2] ON;
   insert into [dbo].[APROBACIONBackUp2] ([apr_id]  ,
	[mod_id] ,
	[o2c_id] ,
	[apr_orden] ,
	[apr_cc_responsable],
	[apr_responsable] ,
	[apr_estado] ,
	[apr_comentarios]  ,
	[apr_fecha],
	[sis_fecha_aprobacion]  ,
	[apn_nivel],
	[prn_Modificar] ,
	[apr_motivoRechazo]  ,
	[apr_motivoAprobacion] ,
	[apr_cc_responsableInicial] ,
	[unn_CodigoUN] 
	)
   select [apr_id], [mod_id] ,
	[o2c_id] ,
	[apr_orden] ,
	[apr_cc_responsable],
	[apr_responsable] ,
	[apr_estado] ,
	[apr_comentarios]  ,
	[apr_fecha],
	[sis_fecha_aprobacion]  ,
	[apn_nivel],
	[prn_Modificar] ,
	[apr_motivoRechazo]  ,
	[apr_motivoAprobacion] ,
	[apr_cc_responsableInicial] ,
	[unn_CodigoUN]  from @dt ;
END
